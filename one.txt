# manual_nn.py
import numpy as np

# 1. Activation Function (Sigmoid) and its Derivative
def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def sigmoid_derivative(x):
    # Note: x is already passed through sigmoid in our loop
    return x * (1 - x) 

# 2. Dataset (XOR Problem)
# Inputs (4 samples, 2 features each)
X = np.array([[0,0], [0,1], [1,0], [1,1]])
# Expected Outputs (4 samples, 1 output each)
y = np.array([[0], [1], [1], [0]])

# 3. Initialization
np.random.seed(42)
input_neurons = 2
hidden_neurons = 2
output_neurons = 1

# Random Weights and Biases
W_hidden = np.random.uniform(size=(input_neurons, hidden_neurons))
b_hidden = np.random.uniform(size=(1, hidden_neurons))

W_output = np.random.uniform(size=(hidden_neurons, output_neurons))
b_output = np.random.uniform(size=(1, output_neurons))

learning_rate = 0.1
epochs = 10000

print("\n--- Training Manual Neural Network ---")
# 4. Training Loop
for epoch in range(epochs):
    
    # --- FORWARD PROPAGATION ---
    # Hidden Layer
    hidden_layer_input = np.dot(X, W_hidden) + b_hidden
    hidden_layer_activation = sigmoid(hidden_layer_input)
    
    # Output Layer
    output_layer_input = np.dot(hidden_layer_activation, W_output) + b_output
    predicted_output = sigmoid(output_layer_input)
    
    # --- BACKPROPAGATION ---
    # Calculate Error
    error = y - predicted_output
    
    # Calculate Gradients for Output Layer
    d_predicted_output = error * sigmoid_derivative(predicted_output)
    
    # Calculate Gradients for Hidden Layer
    error_hidden_layer = d_predicted_output.dot(W_output.T)
    d_hidden_layer = error_hidden_layer * sigmoid_derivative(hidden_layer_activation)
    
    # --- UPDATING WEIGHTS (Gradient Descent) ---
    W_output += hidden_layer_activation.T.dot(d_predicted_output) * learning_rate
    b_output += np.sum(d_predicted_output, axis=0, keepdims=True) * learning_rate
    
    W_hidden += X.T.dot(d_hidden_layer) * learning_rate
    b_hidden += np.sum(d_hidden_layer, axis=0, keepdims=True) * learning_rate

print("Final Predictions after training:")
print(predicted_output)

# pytorch_nn.py
import torch
import torch.nn as nn

# 1. Dataset (XOR Problem)
X = torch.tensor([[0., 0.], [0., 1.], [1., 0.], [1., 1.]])
y = torch.tensor([[0.], [1.], [1.], [0.]])

# 2. Define the Model Structure
model = nn.Sequential(
    nn.Linear(2, 2),  # 2 inputs to 2 hidden neurons
    nn.Sigmoid(),     # Activation
    nn.Linear(2, 1),  # 2 hidden neurons to 1 output
    nn.Sigmoid()      # Activation
)

# 3. Loss Function and Optimizer
loss_fn = nn.MSELoss()
optimizer = torch.optim.SGD(model.parameters(), lr=0.1)

print("\n--- Training PyTorch Neural Network ---")
# 4. Training Loop
epochs = 10000
for epoch in range(epochs):
    # Forward Pass
    y_pred = model(X)
    
    # Calculate Loss
    loss = loss_fn(y_pred, y)
    
    # THE MAGIC HAPPENS HERE (Replaces all math from Manual file)
    optimizer.zero_grad() # Clear old gradients
    loss.backward()       # Calculate new gradients (Backpropagation)
    optimizer.step()      # Update weights

print("Final Predictions after training:")
with torch.no_grad():
    print(model(X))

# topic2_predicate.py

# ---------------------------------------------------------
# 1. THE KNOWLEDGE BASE (Facts we know are 100% true)
# ---------------------------------------------------------
knowledge_base = {
    "human": ["Socrates", "Plato", "Aristotle"],
    "god": ["Zeus", "Athena"],
    "animal": ["Dog", "Cat"]
}

# ---------------------------------------------------------
# 2. PREDICATES (Questions we ask about our facts)
# ---------------------------------------------------------
def is_human(entity):
    # Returns True if the entity exists in the 'human' list
    return entity in knowledge_base["human"]

def is_god(entity):
    # Returns True if the entity exists in the 'god' list
    return entity in knowledge_base["god"]

# ---------------------------------------------------------
# 3. RULES / IMPLICATIONS (The logical "If... Then...")
# ---------------------------------------------------------
def is_mortal(entity):
    # Logic Rule: ∀x (Human(x) → Mortal(x))
    # "For all x, if x is a human, then x is mortal"
    if is_human(entity):
        return True
    # We can add more rules: Animals are also mortal
    elif entity in knowledge_base["animal"]:
        return True
    else:
        return False

def is_immortal(entity):
    # Logic Rule: ∀x (God(x) → Immortal(x))
    # "For all x, if x is a god, then x is immortal"
    if is_god(entity):
        return True
    return False

# ---------------------------------------------------------
# 4. INFERENCE ENGINE (Generating new knowledge)
# ---------------------------------------------------------
print("=== PREDICATE LOGIC INFERENCE ===\n")

# A mix of entities to test our logic on
test_subjects = ["Socrates", "Zeus", "Apollo", "Dog"]

for subject in test_subjects:
    print(f"Testing subject: {subject}")
    
    if is_human(subject):
        print(f"  [Fact] {subject} is a human.")
        if is_mortal(subject):
            print(f"  [Inferred] Therefore, {subject} is mortal.")
            
    elif is_god(subject):
        print(f"  [Fact] {subject} is a god.")
        if is_immortal(subject):
            print(f"  [Inferred] Therefore, {subject} is immortal.")
            
    elif subject in knowledge_base["animal"]:
        print(f"  [Fact] {subject} is an animal.")
        if is_mortal(subject):
            print(f"  [Inferred] Therefore, {subject} is mortal.")

In Artificial Intelligence, First-Order Logic is built on three main pillars: Facts, Rules, and Inference. Here is how the code maps to those concepts:

1. Facts (The Knowledge Base)

The AI Concept: In logic, a predicate is a property or relation, like Human(Socrates). This simply means "Socrates is a human."

Premise 1: If P, then Q. (If human, then mortal)

what is predicate and fol?

It is very smart to ask this! You've seen the code, but for an exam, you need to know how to explain the actual theory behind those terms.

Here is the simple, plain-English breakdown of what Predicates and First-Order Logic (FOL) actually are in Artificial Intelligence.

1. What is a Predicate? (Predicate Logic)
A Predicate is essentially a statement or a question about an object (or multiple objects) that evaluates to either True or False.

Think of it like a property or a relationship. Instead of just saying "Socrates is a man" (Propositional Logic), Predicate logic breaks it down into an object and its property.

Syntax: Predicate(Object)

Unary Predicate (One Object): Describes a property.

Human(Socrates) → "Socrates is a human." (True)

Even(5) → "5 is an even number." (False)

Binary/n-ary Predicate (Multiple Objects): Describes a relationship.

Brother(Thor,Loki) → "Thor is the brother of Loki." (True)

Greater(3,10) → "3 is greater than 10." (False)

In your code: The functions like is_cs_major("Alice") or has_completed("Alice", "CS101") are your predicates. They look at objects and return True or False.

2. What is FOL? (First-Order Logic)
First-Order Logic (FOL) is simply Predicate Logic plus Quantifiers.

While Predicate Logic lets you talk about specific objects (like Socrates or Alice), FOL lets you make sweeping rules about groups of objects without having to name every single one of them. It does this using two powerful mathematical symbols:

A. The Universal Quantifier (∀) - "For All"
This means the rule applies to every single thing in your universe.

Math: ∀x(Human(x)→Mortal(x))

English: "For ALL x, if x is a Human, then x is Mortal." (All humans are mortal).

Code Equivalent: Python's all() function.

B. The Existential Quantifier (∃) - "There Exists"
This means the rule only needs to be true for at least one thing in your universe.

Math: ∃x(Bird(x)∧CannotFly(x))

English: "There EXISTS at least one x, where x is a Bird AND x cannot fly." (Some birds cannot fly, like penguins).

Code Equivalent: Python's any() function, or checking if a list length is > 0.

Summary: The Difference Between Them
To put it in perspective for tomorrow's exam:

Predicate Logic gives you the vocabulary to state basic facts: Student(Alice), Course(AI_401).

First-Order Logic gives you the grammar to create complex, intelligent rules about those facts: ∀x(Student(x)∧Passed(x,AI_401)→Graduate(x)).

# ---------------- activation functions ----------------

def relu(x):

    return [max(0, i) for i in x]



def relu_deriv(x):

    return [1 if i > 0 else 0 for i in x]



def softmax(x):

    exps = [math.exp(i - max(x)) for i in x]

    s = sum(exps)

    return [i / s for i in exps]



def loss_fn(p, y):

    return -math.log(p[y] + 1e-9)





# ---------------- dataset ----------------

def gen_data(n=100, f=10):

    X, Y = [], []

    for _ in range(n):

        x = [random.uniform(-1, 1) for _ in range(f)]

        y = 1 if sum(x) > 0 else 0

        X.append(x)

        Y.append(y)

    return X, Y

# ---------------- shuffle helper ----------------

def shuffle_data(X, Y):

    combined = list(zip(X, Y))

    random.shuffle(combined)

    X, Y = zip(*combined)

    return list(X), list(Y)



# ---------------- weight init (better than uniform) ----------------

def rand_mat(r, c):

    return [[random.gauss(0, 0.1) for _ in range(c)] for _ in range(r)]



W1 = rand_mat(10, 5)

W2 = rand_mat(5, 2)



lr = 0.01

epochs = 100



# ---------------- training ----------------

for epoch in range(epochs):



    # shuffle every epoch (IMPORTANT)

    X, Y = shuffle_data(X, Y)



    total_loss = 0

    correct = 0



    for x, y in zip(X, Y):



        # ---------- forward ----------

        h = [sum(x[i] * W1[i][j] for i in range(10)) for j in range(5)]

        h_relu = relu(h)



        o = [sum(h_relu[i] * W2[i][j] for i in range(5)) for j in range(2)]

        p = softmax(o)



        loss = loss_fn(p, y)

        total_loss += loss



        pred = p.index(max(p))

        if pred == y:

            correct += 1



        # ---------- backward ----------

        grad_o = p[:]

        grad_o[y] -= 1

        for i in range(5):

            for j in range(2):

                W2[i][j] -= lr * grad_o[j] * h_relu[i]

        grad_h = [0] * 5

        for i in range(5):

🎲 What is a "Seed" in Programming?
To understand this line, you first need to know how computers generate random numbers.

Computers are purely logical machines; they can't do anything truly "random" on their own. Instead, they use complex math formulas to generate Pseudo-Random Numbers. These formulas need a starting point to begin their calculations. That starting point is called the seed.

Here is how random.seed() controls that starting point:

1. Fixed Seed (For Reproducibility)
If you give the seed a specific integer, like random.seed(42), the math formula starts from the exact same point every single time you run the script.

Result: The "random" numbers generated (like your initialized weights or shuffled data) will be 100% identical every time you click Run.

Why use this? It is crucial for debugging Neural Networks. If your network suddenly breaks, you want to be able to run it again with the exact same data and weights to figure out what went wrong.

2. None Seed (For True Unpredictability)
When you use random.seed(None) (which is what your code has), you are telling Python: "Don't use a fixed starting point. Use the current system time (down to the millisecond) as the seed."

Result: Because the system time is always changing, the math formula gets a different starting point every time you run the script. Your weights and data shuffling will be completely different each run.

# ---------------- shuffle helper ----------------

def shuffle_data(X, Y):

    combined = list(zip(X, Y))

    random.shuffle(combined)

    X, Y = zip(*combined)

    return list(X), list(Y)

This block of code is a clever little helper function designed to shuffle your dataset.

In machine learning, if you feed a neural network the exact same data in the exact same order during every epoch (training loop), it can sometimes memorize the order of the data rather than actually learning the rules. Shuffling the data prevents this and helps the network generalize better.

However, you have a problem when shuffling: your inputs (X) and your answers (Y) are in two separate lists. If you shuffle X but forget to shuffle Y in the exact same way, your inputs will suddenly point to the wrong answers, and your network will learn complete garbage!

Here is how this function brilliantly solves that problem using Python's zip command:

Line-by-Line Breakdown
1. "Gluing" them together

Python
combined = list(zip(X, Y))
The zip() function takes two lists and "zips" them together like a jacket. It takes the first item of X and pairs it with the first item of Y, creating a single package (a tuple).

Before: X = [DataA, DataB], Y = [LabelA, LabelB]

After: combined = [(DataA, LabelA), (DataB, LabelB)]

2. Shuffling the pairs

Python
random.shuffle(combined)
Now that the data and its correct label are securely glued together in a single package, random.shuffle() can safely mix up the order of the list. Even though the order of the rows changes, DataA is still permanently attached to LabelA.

3. "Unzipping" them back apart

Python
X, Y = zip(*combined)
In Python, using zip(*...) with the asterisk (*) is the trick to unzip a list of pairs. It rips the packages back open, putting all the data back into an X pile and all the labels back into a Y pile. The order is now totally randomized, but they still perfectly line up with each other!

4. Returning the lists

Python
return list(X), list(Y)
Because the unzipping process creates immutable "tuples" (lists you can't edit), this final line just converts them back into standard Python lists so your neural network can easily use them in the next training loop.

2. Activation Functions & Loss
Python
def relu(x):
    return [max(0, i) for i in x]
How it works: Loops through a list x. If a number i is less than 0, max(0, i) replaces it with 0. If it's greater than 0, it stays the same.

What to learn: This is the ReLU (Rectified Linear Unit) function. It introduces "non-linearity," allowing the network to learn complex patterns instead of just straight lines.

Python
def relu_deriv(x):
    return [1 if i > 0 else 0 for i in x]
How it works: Loops through list x. Returns a 1 if the original number was positive, and a 0 if it was negative or zero.

What to learn: This is the derivative (slope) of ReLU. During backpropagation, a 0 stops the network from updating weights for a "dead" neuron, while a 1 allows the learning signal to pass through.

Python
def softmax(x):
    exps = [math.exp(i - max(x)) for i in x]
    s = sum(exps)
    return [i / s for i in exps]
How it works: Subtracts the highest number in the list from everything (i - max(x)) to prevent mathematical overflow. Then it calculates the exponential (e 
x
 ) for each number, sums them up, and divides each by the sum.

What to learn: Softmax turns raw, messy network outputs into neat percentages (probabilities) that add up exactly to 1.0.

Python
def loss_fn(p, y):
    return -math.log(p[y] + 1e-9)
How it works: Takes the network's probability list p and the correct answer index y. It finds the probability the network assigned to the correct answer, adds a tiny number (1e-9) so it never tries to calculate the log of exactly zero, and calculates the negative logarithm.

What to learn: This is Cross-Entropy Loss. It heavily penalizes the network if it is highly confident in the wrong answer.

import itertools
import re

def analyze_logic(expr):
    print(f"Original Expression: {expr}")
    
    # 1. PARSING: Convert standard logic operators to Python logic
    # We replace math symbols with Python's equivalent boolean operators
    parsed_expr = expr.replace('<->', ' == ') # Biconditional (IFF)
    parsed_expr = parsed_expr.replace('->', ' <= ')   # Implication (IMPLIES)
    # The user should use 'and', 'or', 'not' for the rest.
    
    # Extract unique variables (e.g., 'p', 'q', 'r') using regex
    # Looks for standalone lowercase letters
    vars_list = sorted(list(set(re.findall(r'\b[a-z]\b', parsed_expr))))
    
    if not vars_list:
        print("No variables found in expression.")
        return

    # 2. Generate all Truth Assignments (True/False combos)
    combinations = list(itertools.product([True, False], repeat=len(vars_list)))
    
    # Setup Truth Table UI
    header = " | ".join(vars_list) + " | Result"
    print("-" * len(header))
    print(header)
    print("-" * len(header))
    
    results = []
    
    # 3. Evaluate the expression for every assignment
    for combo in combinations:
        # Map variables to current truth values (e.g., {'p': True, 'q': False})
        env = dict(zip(vars_list, combo))
        
        try:
            # eval() safely runs the string as actual Python code
            result = eval(parsed_expr, {}, env)
            results.append(result)
            
            # Format row output (Print 'T' / 'F' instead of True/False)
            row_str = " | ".join(['T' if env[v] else 'F' for v in vars_list])
            res_str = 'T' if result else 'F'
            print(f"{row_str} |   {res_str}")
            
        except Exception as e:
            print(f"Error evaluating expression: {e}")
            return

    print("-" * len(header))
    
    # 4. Check status (Tautology / Contradiction / Contingent)
    if all(results):
        print("Conclusion: TAUTOLOGY (Always True)")
    elif not any(results):
        print("Conclusion: CONTRADICTION (Always False)")
    else:
        print("Conclusion: CONTINGENT (Mixed Results)")
    print("\n")

        print(f"{row_str} |   {res_str}")

Question 1: Describe components of a neural network and discuss the similarities of a neural network with the human brain. (6 Marks)
Part A: Components of an Artificial Neural Network (ANN)
An ANN is built using several foundational components that process data:

Neurons (Nodes): The basic mathematical units that hold a number. They are arranged in layers (Input layer, Hidden layers, and Output layer).

Weights: Connections between neurons. They represent the "strength" or importance of a particular input. When data passes from one neuron to another, it is multiplied by the weight.

Biases: An extra constant value added to the sum of inputs and weights before passing through the activation function. It allows you to shift the activation function curve up or down.

Activation Functions: Mathematical equations (like ReLU, Sigmoid, or Step functions) attached to each neuron. They decide whether a neuron should "fire" (pass its signal forward) based on the calculated input, introducing non-linearity to the network so it can learn complex patterns.

Part B: Similarities with the Human Brain
Artificial Neural Networks were originally inspired by biological brains. The key analogies include:

Nodes vs. Biological Neurons: An artificial node acts like a brain cell (neuron), receiving signals, processing them, and transmitting a result.

Weights vs. Synapses: In the brain, synapses connect neurons and can grow stronger or weaker based on usage. In an ANN, weights represent this synaptic strength.

Activation vs. Action Potential: A biological neuron fires an electrical impulse (action potential) only when the incoming signals cross a certain threshold. An artificial activation function mimics this by mathematically restricting when and how strongly a node passes data forward.

Distributed Knowledge: Both systems do not store "memory" in one specific central location (like a standard computer hard drive). Instead, knowledge is distributed across the entire network as the collective strengths of all connections.

Question 2: What does "learning" mean referring to neural nets? (4 Marks)
In the context of artificial neural networks, "learning" is the mathematical process of automatically adjusting the network's internal parameters (weights and biases) to minimize errors. A network doesn't start out knowing how to solve a problem; its initial weights are usually random. "Learning" involves:

Experience: Feeding the network data (training examples).

Evaluation: Calculating how wrong the network's prediction was compared to the actual target using a Loss Function.

Adaptation: Using algorithms like Gradient Descent and Backpropagation to push the error backward through the network, tweaking the weights and biases slightly so that the next time it sees similar data, it makes a more accurate prediction.

Learning is considered complete when the network's loss stops decreasing, meaning it has generalized the underlying patterns of the data.

Question 4: Explain forward propagation and self-organization. (3 Marks)
Forward Propagation:
This is the process of pushing input data through the neural network from the input layer, through the hidden layers, and out to the output layer to generate a prediction. During this phase, data flows strictly in one direction (forward).
At each neuron, the inputs are multiplied by their respective weights, summed together with the bias, and passed through an activation function to become the input for the next layer. No learning or weight updating happens during forward propagation; it is purely the network making a "guess."

Self-Organization:
While forward propagation is standard for supervised learning, self-organization refers to a specific type of unsupervised learning. In a self-organizing neural network (like a Kohonen Self-Organizing Map), neurons "compete" with each other to respond to specific inputs.
Without any target labels to guide them, the network's nodes mathematically arrange themselves based on the input data's topology. Similar inputs will trigger neurons that are physically close to each other in the network. It "organizes itself" to represent the hidden patterns in the data without human intervention or backpropagated error signals.
